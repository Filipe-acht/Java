Métodos
	Anatomia:
		* primeira letra minuscula
		* Tipos de retorno -> void, String, Int...
		* pode ter o mesmo nome contando que os parametros sejam diferentes
		
		TipoRetorno nomeMetodo(parametros de entrada){
		
			Algoritmo
			
			return -> retornar o valor
		}
		
		tipo nomeMetodo(tipo1 param1, tipo2 param2, ...){
			// corpo
		}

		int somar(int a, int b){

		}
